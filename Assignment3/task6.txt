function studentResults(int $m) {
    if ($m >= 75)
        echo "Distinction";
    else if ($m >= 50 && $m < 75)
        echo "Pass";
    else if ($m >= 40 && $m <= 49)
        echo "Supplementary";
    else
        echo "Fail";
}
